    \section{导数表}   
      % \[(C)~'&=0 \]
      % \[(x^\mu)~'=\mu x^{\mu-1}\]
      % \[(\sin x)~'=\cos x\]
      % \[(\cos x)~'=-\sin x\]
      % \[(\tan x)~'=\sec^2 x\]
      % \[(\cot x)~'=-\csc^2 x\]
      % \[(\sec x)~'=\sec x\tan x\]
      % \[(\csc x{})~'=-\csc x\cot x\]
      % \[(a^x)~'=a^x\ln_{}{a}\quad(a> 0,a\neq1)\]
      % \[(e^x)~'=e^x\]
      % \[(\log_{a}{x})'=\frac{1}{x\ln_{}{a}}\quad(a> 0,a\neq1)\]
      % \[(\ln_{}{x})'=\frac{1}{x} \]
      \begin{equation}  
        \begin{split}   % split为分割网
          (C)~'&=0\\
          (x^\mu)~'&=\mu x^{\mu-1}\\
          (\sin x)~'&=\cos x\\
          (\cos x)~'&=-\sin x\\
          (\tan x)~'&=\sec^2 x\\
          (\cot x)~'&=-\csc^2 x\\
          (\sec x)~'&=\sec x\tan x\\
          (\csc x{})~'&=-\csc x\cot x\\
          (a^x)~'&=a^x\ln_{}{a}\quad(a> 0,a\neq1)\\
          (e^x)~'&=e^x\\
          (\log_{a}{x})'&=\frac{1}{x\ln_{}{a}}\quad(a> 0,a\neq1)\\
          (\ln_{}{x})'&=\frac{1}{x} \\
        \end{split}
    \end{equation}
